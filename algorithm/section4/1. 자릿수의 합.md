function solution1(arr) {
  let answer = 0;
  let maxSum = Number.MIN_SAFE_INTEGER;

  for (const num of arr) {
    let sum = 0;
    let n = num;

    while (n > 0) {
      sum += n % 10;
      n = Math.floor(n / 10);
    }

    if (sum === maxSum && answer < num) answer = num;
    if (sum > maxSum) {
      maxSum = sum;
      answer = num;
    }
  }

  return answer;
}
