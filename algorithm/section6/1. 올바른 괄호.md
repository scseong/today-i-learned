Solution 1: Stack

```js
function solution1(s) {
  let stack = [];

  for (const x of s) {
    if (x === "(") stack.push(x);
    else {
      if (stack.length === 0) return "NO";
      stack.pop();
    }
  }

  return stack.length > 0 ? "NO" : "YES";
}
```

Solution 2: Counting

```js
function solution2(s) {
  let count = 0;

  for (const x of s) {
    if (x === "(") count++;
    else {
      count--;
      if (count < 0) return "NO";
    }
  }

  return count === 0 ? "YES" : "NO";
}
```
